// 渲染博客卡片（只使用blog-container作为卡片容器） function renderBlogs(filterTag = null) { if (!blogContainer) return; // 清空容器并显示 blogContainer.innerHTML = ''; blogContainer.style.display = 'block'; // 过滤博客 const filteredBlogs = filterTag ? allBlogs.filter(blog => blog.tags.includes(filterTag)) : allBlogs; // 处理无结果情况 if (filteredBlogs.length === 0) { noResultsEl?.style.display = 'block'; return; } else { noResultsEl?.style.display = 'none'; } // 创建统一的卡片容器（直接作为blog-container的子元素） const cardsContainer = document.createElement('div'); cardsContainer.className = 'cards-container'; // 复用原有的网格布局样式 // 遍历所有过滤后的博客，创建卡片并添加到容器 filteredBlogs.forEach(article => { cardsContainer.appendChild(createBlogCard(article)); }); // 将卡片容器添加到blog-container blogContainer.appendChild(cardsContainer); }
